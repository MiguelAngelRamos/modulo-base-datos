### 1.1 Creación del Esquema y Tablas (DDL)

```sql
CREATE SCHEMA IF NOT EXISTS electronica_store;

-- 1. Categorías
CREATE TABLE electronica_store.categories (
    id INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    description TEXT,
    CONSTRAINT uq_categories_name UNIQUE (name)
);

-- 2. Productos (3FN: Relacionado a categoría)
CREATE TABLE electronica_store.products (
    id INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    category_id INT,
    name VARCHAR(100) NOT NULL,
    brand VARCHAR(50),
    price DECIMAL(10,2) NOT NULL,
    stock INT DEFAULT 0,
    CONSTRAINT fk_products_category_id FOREIGN KEY (category_id) 
        REFERENCES electronica_store.categories(id),
    CONSTRAINT chk_products_price CHECK (price > 0)
);

-- 3. Clientes
CREATE TABLE electronica_store.customers (
    id INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    full_name VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL,
    country VARCHAR(50),
    CONSTRAINT uq_customers_email UNIQUE (email)
);

-- 4. Ventas (Orders)
CREATE TABLE electronica_store.sales (
    id INT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    customer_id INT,
    sale_date DATE DEFAULT CURRENT_DATE,
    total_amount DECIMAL(10,2) DEFAULT 0,
    CONSTRAINT fk_sales_customer_id FOREIGN KEY (customer_id) 
        REFERENCES electronica_store.customers(id)
);

```
